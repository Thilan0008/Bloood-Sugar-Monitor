String getCategory(double before, double after) {
  if (before < 100 && after < 140) {
    return 'Normal';
  } else if (before < 126 && after < 180) {
    return 'Prediabetes';
  } else if (before >= 126 && before <= 180 && after >= 180 && after <= 200) {
    return 'Prediabetes';
  } else {
    return 'Diabetes';
  }
}
